# 核心矛盾：
① 下游根据 nginx server_name abc.com 做的分发，所以必须设置 proxy_set_header Host abc.com;
② 后端用这个 Host 生成重定向 URL → 导致 301 跳转到 http://abc.com/... → 客户端无法解析

问题1: 为什么 会出现 301 的重定向？
回答1: 很多 Web 框架（如 Spring MVC、Flask、Django）对 末尾斜杠敏感：
        /login → 被视为“文件”
        /login/ → 被视为“目录”
    如果路由定义为 /login/，但用户访问 /login，框架会自动 Location 301 到 /login/。这是正常行为。


# 解决方案：

server {
    listen 18186;

    access_log /var/log/nginx/access-$server_port.log main;

    location / {
        proxy_set_header Host abc.com;

        proxy_set_header X-Forwarded-Host $http_host;
        proxy_set_header X-Forwarded-Port $server_port;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_buffering off;
        proxy_http_version 1.1;
        proxy_connect_timeout 60s;
        proxy_read_timeout 300s;

        proxy_redirect ~^http://abc\.com(?::\d+)?(/.*)$ http://$host:$server_port$1;
        proxy_redirect ~^https://abc\.com(?::\d+)?(/.*)$ http://$host:$server_port$1;  # 如果有 HTTPS 也加

        proxy_pass http://10.205.229.104;
    }

    error_page 500 502 503 504 /50x.html;
    location = /50x.html {
        root html;
    }
}
